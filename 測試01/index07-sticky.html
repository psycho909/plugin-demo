function countDown(targetTime) {
                            const now = new Date().getTime();
                            const targetDate = new Date(now + targetTime);
                          
                            setInterval(() => {
                              const remainingTime = targetDate.getTime() - new Date().getTime();
                              if (remainingTime <= 0) {
                                alert("Time's up!");
                                return;
                              }
                          
                              const days = Math.floor(remainingTime / (1000 * 60 * 60 * 24));
                              const hours = Math.floor((remainingTime % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                              const minutes = Math.floor((remainingTime % (1000 * 60 * 60)) / (1000 * 60));
                              const seconds = Math.floor(remainingTime / 1000);
                          
                              alert(`${days} days, ${hours} hours, ${minutes} minutes, and ${seconds} seconds`);
                            }, 1000);
                          }
                          
                          countDown(60 * 60 * 1000); // This will start a countdown for 1 hour (3600 seconds)
